#### Supabase Edge Function: `generate-sitemap`

**File:** `/supabase/functions/generate-sitemap/index.ts`

```typescript
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
  'Content-Type': 'application/xml',
};

serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL') ?? '',
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? ''
    );

    const url = new URL(req.url);
    const type = url.searchParams.get('type') || 'index';

    if (type === 'index') {
      return generateSitemapIndex();
    } else if (type === 'static') {
      return generateStaticSitemap();
    } else if (type === 'blog') {
      return generateBlogSitemap(supabase);
    } else if (type === 'profiles') {
      return generateProfilesSitemap(supabase);
    } else if (type === 'pages') {
      return generatePagesSitemap(supabase);
    }

    return new Response('Invalid type', { status: 400 });
  } catch (error) {
    return new Response(JSON.stringify({ error: error.message }), {
      status: 500,
      headers: { 'Content-Type': 'application/json' },
    });
  }
});

function generateSitemapIndex() {
  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<sitemapindex xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
  <sitemap>
    <loc>https://agentbio.net/api/sitemap?type=static</loc>
    <lastmod>${new Date().toISOString()}</lastmod>
  </sitemap>
  <sitemap>
    <loc>https://agentbio.net/api/sitemap?type=blog</loc>
    <lastmod>${new Date().toISOString()}</lastmod>
  </sitemap>
  <sitemap>
    <loc>https://agentbio.net/api/sitemap?type=profiles</loc>
    <lastmod>${new Date().toISOString()}</lastmod>
  </sitemap>
  <sitemap>
    <loc>https://agentbio.net/api/sitemap?type=pages</loc>
    <lastmod>${new Date().toISOString()}</lastmod>
  </sitemap>
</sitemapindex>`;

  return new Response(xml, { headers: corsHeaders });
}

    { url: 'privacy', priority: '0.3', changefreq: 'yearly' },

  ];

  const urls = staticPages.map(page => `

  return new Response(xml, { headers: corsHeaders });
}

async function generateBlogSitemap(supabase: any) {
  const { data: articles, error } = await supabase
    .from('articles')
    .select('slug, updated_at, published_at')
    .eq('is_published', true)
    .order('published_at', { ascending: false });

  if (error) throw error;

  const urls = articles.map((article: any) => `
  <url>
    <loc>https://agentbio.net/blog/${article.slug}</loc>
    <lastmod>${new Date(article.updated_at).toISOString()}</lastmod>
    <changefreq>monthly</changefreq>
    <priority>0.7</priority>
  </url>`).join('');

  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
${urls}
</urlset>`;

  return new Response(xml, { headers: corsHeaders });
}

async function generateProfilesSitemap(supabase: any) {
  const { data: profiles, error } = await supabase
    .from('profiles')
    .select('username, updated_at')
    .not('username', 'is', null)
    .limit(10000); // Limit for sitemap size

  if (error) throw error;

  const urls = profiles.map((profile: any) => `
  <url>
    <loc>https://agentbio.net/@${profile.username}</loc>
    <lastmod>${new Date(profile.updated_at).toISOString()}</lastmod>
    <changefreq>weekly</changefreq>
    <priority>0.6</priority>
  </url>`).join('');

  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
${urls}
</urlset>`;

  return new Response(xml, { headers: corsHeaders });
}

async function generatePagesSitemap(supabase: any) {
  const { data: pages, error } = await supabase
    .from('custom_pages')
    .select('slug, updated_at')

  if (error) throw error;

  const urls = pages.map((page: any) => `
  <url>
    <loc>https://agentbio.net/p/${page.slug}</loc>
    <lastmod>${new Date(page.updated_at).toISOString()}</lastmod>
    <changefreq>monthly</changefreq>
    <priority>0.5</priority>
  </url>`).join('');

  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
${urls}
</urlset>`;

  return new Response(xml, { headers: corsHeaders });
}
```
